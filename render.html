import fs from "fs"; import path from "path"; import { HealFlow } from "./types"; /** * Self-Healing Flow Dashboard * --------------------------- * Renders the dark/light themed dashboard (resilience ring, tier-distribution * bar, stat cards, grouped timeline, expandable detail rows) from the live * `flows[]` array. Every figure is derived from the data — nothing hardcoded. */ const esc = (s?: string | number | null) => String(s ?? "—") .replace(/&/g, "&") .replace(//g, ">") .replace(/"/g, """); const displayNames: Record = { continue_shopping: "Dismiss Welcome Popup", search_toggle: "Open Product Search", search_input: "Enter Search Criteria", dress_product: "Select Product", dress_size: "Choose Size", add_to_bag: "Add Product To Bag", mini_cart_open: "Open Shopping Bag", verify_cart: "Verify Cart Contents", }; const TIER_META = { db: { label: "DB Primary", color: "#2DD4A7" }, smart: { label: "Smart Locator", color: "#F4B740" }, ai: { label: "AI Healed", color: "#FF6B6B" }, unresolved: { label: "Unresolved", color: "#6B7A99" }, } as const; type TierKey = keyof typeof TIER_META; function pct(n: number, total: number): number { return total === 0 ? 0 : Math.round((n / total) * 1000) / 10; } export function renderHtml(flows: HealFlow[], outDir: string): string { const total = flows.length; const count = (t: string) => flows.filter((f) => f.finalTier === t).length; const dbN = count("db"); const smartN = count("smart"); const aiN = count("ai"); const unresolvedN = flows.filter((f) => !f.finalTier).length; const healedN = flows.filter((f) => f.healed).length; const resolvedN = total - unresolvedN; const resilience = pct(resolvedN, total); const totalMs = flows.reduce( (a, f) => a + (f.steps?.reduce((s, st) => s + (st.durationMs ?? 0), 0) ?? 0), 0 ); const distOrder: TierKey[] = ["db", "smart", "ai", "unresolved"]; const distSegments = distOrder .map((k) => { const n = k === "unresolved" ? unresolvedN : k === "db" ? dbN : k === "smart" ? smartN : aiN; const w = pct(n, total); if (w === 0) return ""; return `
${ w >= 8 ? `${n}` : "" }
`; }) .join(""); const distLegend = distOrder .map((k) => { const n = k === "unresolved" ? unresolvedN : k === "db" ? dbN : k === "smart" ? smartN : aiN; return `
  • ${TIER_META[k].label}${n}
    • `; }) .join(""); const rows = flows .map((f, i) => { const tier = (f.finalTier ?? "unresolved") as TierKey; const meta = TIER_META[tier] ?? TIER_META.unresolved; const displayName = displayNames[f.elementName] ?? f.elementName; const stepRows = (f.steps ?? []) .map( (s, si) => ` ${(s as any).order ?? si + 1} ${esc(s.tier)} ${esc(s.action)} ${esc(s.status)} ${esc(s.locator)} ${s.durationMs != null ? s.durationMs + " ms" : "—"} ` ) .join(""); return ` ${String(i + 1).padStart(2, "0")} ${esc(displayName)} ${meta.label} ${f.healed ? `Healed` : `—`} ${esc(f.finalLocator)} ▾ ${stepRows || ``}
    #TierActionStatusLocatorDuration
    No steps recorded
    `; }) .join(""); const generated = new Date().toLocaleString(); const html = ` Self-Healing Flow Report
    Knitwell · Self-Healing

    Flow Resolution Dashboard

    Live results from the three-tier fallback chain — DB primary → smart locators → AI healing
    Generated
    ${esc(generated)}
    ${resilience}%
    Resilience

    Tier Distribution · ${total} elements

    ${distSegments || '
    '}
      ${distLegend}
    ${total}
    Total Elements
    ${dbN}
    DB Resolved
    ${smartN}
    Smart Healed
    ${aiN}
    AI Healed
    ${unresolvedN}
    Unresolved
    ${resilience}%
    Resilience Score
    Element Detail · click a row to expand its tier trace
    ${rows || ''}
    #ElementFinal TierHealedFinal Locator
    No elements captured in this run.
    ${resolvedN}/${total} resolved · ${healedN} healed · ${totalMs} ms total execution time
    `; fs.writeFileSync(path.join(outDir, "report-data.json"), JSON.stringify(flows, null, 2)); const out = path.join(outDir, "index.html"); fs.writeFileSync(out, html); console.log("Dashboard created:", out); return out; }
